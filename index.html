// Supabase Edge Function: regint-manage-user // Deploy: supabase functions deploy regint-manage-user --no-verify-jwt // Segreti richiesti: SERVICE_ROLE_KEY // // Actions disponibili: // check_email → verifica se l'email esiste già in Supabase Auth // create → crea profilo; se l'email non esiste in Auth la crea prima // update → aggiorna profilo e (opzionalmente) password Auth // delete_profile→ elimina SOLO il profilo (non l'utente Auth) import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', 'Access-Control-Allow-Methods': 'POST, OPTIONS', } const RUOLI_VALIDI = ['admin', 'utente'] function json(data: unknown, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...CORS, 'Content-Type': 'application/json' }, }) } Deno.serve(async (req) => { if (req.method === 'OPTIONS') return new Response('ok', { headers: CORS }) try { const SUPABASE_URL = Deno.env.get('SUPABASE_URL')! const SERVICE_KEY = Deno.env.get('SERVICE_ROLE_KEY')! const ANON_KEY = Deno.env.get('SUPABASE_ANON_KEY')! // ── Verifica chiamante autenticato e admin ────────────────── const authHeader = req.headers.get('Authorization') if (!authHeader) throw new Error('Non autorizzato') const anonClient = createClient(SUPABASE_URL, ANON_KEY, { global: { headers: { Authorization: authHeader } }, }) const { data: { user: caller }, error: authErr } = await anonClient.auth.getUser() if (authErr || !caller) throw new Error('Non autorizzato') const { data: callerProfile, error: profErr } = await anonClient .from('regint_profili') .select('ruolo') .eq('id', caller.id) .single() if (profErr || callerProfile?.ruolo !== 'admin') throw new Error('Accesso riservato agli amministratori') const admin = createClient(SUPABASE_URL, SERVICE_KEY) const body = await req.json() const { action } = body // ══════════════════════════════════════════════════════════ // CHECK_EMAIL // ══════════════════════════════════════════════════════════ if (action === 'check_email') { const { email } = body if (!email) throw new Error('Email obbligatoria') const { data: list, error } = await admin.auth.admin.listUsers() if (error) throw new Error(error.message) const existing = list.users.find(u => u.email === email.toLowerCase().trim()) return json({ ok: true, exists: !!existing, user_id: existing?.id ?? null }) } // ══════════════════════════════════════════════════════════ // CREATE // ══════════════════════════════════════════════════════════ if (action === 'create') { const { email, password, nome, cognome, ruolo } = body if (!email) throw new Error('Email obbligatoria') if (!RUOLI_VALIDI.includes(ruolo)) throw new Error(`Ruolo non valido. Valori ammessi: ${RUOLI_VALIDI.join(', ')}`) const emailNorm = email.toLowerCase().trim() const { data: list, error: listErr } = await admin.auth.admin.listUsers() if (listErr) throw new Error(listErr.message) const existing = list.users.find(u => u.email === emailNorm) let userId: string if (existing) { userId = existing.id const { data: existingProfile } = await admin .from('regint_profili') .select('id') .eq('id', userId) .maybeSingle() if (existingProfile) throw new Error('Questo utente ha già un profilo nel Registro Interventi') } else { if (!password) throw new Error('Password obbligatoria per i nuovi utenti') if (password.length < 8) throw new Error('Password minimo 8 caratteri') const { data: newUser, error: createErr } = await admin.auth.admin.createUser({ email: emailNorm, password, email_confirm: true, user_metadata: { nome, cognome, ruolo }, }) if (createErr) throw new Error(createErr.message) userId = newUser.user.id } const { error: upsertErr } = await admin .from('regint_profili') .upsert({ id: userId, nome: (nome || '').trim(), cognome: (cognome || '').trim(), email: emailNorm, ruolo: ruolo || 'utente', attivo: true, }, { onConflict: 'id' }) if (upsertErr) throw new Error(upsertErr.message) return json({ ok: true, id: userId, // retrocompatibilità con il vecchio campo "id" user_id: userId, auth_created: !existing, }) } // ══════════════════════════════════════════════════════════ // UPDATE // ══════════════════════════════════════════════════════════ if (action === 'update') { const { user_id, nome, cognome, email, ruolo, attivo, password } = body if (!user_id) throw new Error('user_id obbligatorio') if (ruolo && !RUOLI_VALIDI.includes(ruolo)) throw new Error(`Ruolo non valido. Valori ammessi: ${RUOLI_VALIDI.join(', ')}`) const profilePayload: Record = {} if (nome !== undefined) profilePayload.nome = nome.trim() if (cognome !== undefined) profilePayload.cognome = cognome.trim() if (email !== undefined) profilePayload.email = email.toLowerCase().trim() if (ruolo !== undefined) profilePayload.ruolo = ruolo if (attivo !== undefined) profilePayload.attivo = attivo if (Object.keys(profilePayload).length) { const { error: updErr } = await admin .from('regint_profili') .update(profilePayload) .eq('id', user_id) if (updErr) throw new Error(updErr.message) } if (password) { if (password.length < 8) throw new Error('Password minimo 8 caratteri') const { error: pwdErr } = await admin.auth.admin.updateUserById(user_id, { password }) if (pwdErr) throw new Error(pwdErr.message) } return json({ ok: true }) } // ══════════════════════════════════════════════════════════ // DELETE_PROFILE — rimuove il profilo, NON l'utente Auth // ══════════════════════════════════════════════════════════ if (action === 'delete_profile') { const { user_id } = body if (!user_id) throw new Error('user_id obbligatorio') if (user_id === caller.id) throw new Error('Non puoi eliminare il tuo stesso profilo') const { error: delErr } = await admin .from('regint_profili') .delete() .eq('id', user_id) if (delErr) throw new Error(delErr.message) return json({ ok: true }) } throw new Error(`Azione non riconosciuta: ${action}`) } catch (e: unknown) { const msg = e instanceof Error ? e.message : 'Errore sconosciuto' return json({ ok: false, error: msg }, 400) } })
